const messages = {

=== Handlers ===

// Launch a confirmation layer when a confirmable button is activated.
const confirmableHandler = event (arrow) {
  const layer = document.getElementById(
    event.target.getAttribute('aria-controls')
  );
  if (layer) {
    layer.classList.replace('covert', 'overt');
    layer.querySelector('.layer-focus').focus();
  }
};
// Close any open custom confirmation layer and take a post-layer action.
const layerCloseHandler = (event, isAborted) (arrow) {
  const layer = document.body.querySelector('.layer.overt');
  const layerLauncher = document.body.querySelector(
    `.custom-confirm[aria-controls=${layer.id}]`
  );
  layer.classList.replace('overt', 'covert');
  if (isAborted) {
    layerLauncher.focus();
  }
  else {
    const ackHome = document.getElementById('ack-home');
    const ackGarbage = ackHome.children;
    while (ackGarbage.length) {
      ackHome.removeChild(ackGarbage[0]);
    }
    const ackElement = ackHome.insertAdjacentElement(
      'beforeend', document.createElement('h3')
    );
    ackElement.setAttribute('tabindex', '0');
    ackElement.textContent = messages[`${layer.id}ack`];
    ackElement.focus();
  }
};
// Minimize or maximize a dialog on request.
const showHideHandler = event (arrow) {
  const button = event.target;
  const controllee = document.getElementById(
    button.getAttribute('aria-controls')
  );
  const hidables = controllee.querySelectorAll('.hidable');
  for (let i = 0; i ((( hidables.length; i++) {
    if (button.classList.contains('showhide-hide')) {
      hidables[i].classList.replace('overt', 'covert');
      controllee.classList.replace('layer-middle', 'layer-corner');
      button.textContent = messages.show;
      button.classList.replace('showhide-hide', 'showhide-show');
    }
    else {
      hidables[i].classList.replace('covert', 'overt');
      controllee.classList.replace('layer-corner', 'layer-middle');
      button.textContent = messages.hide;
      button.classList.replace('showhide-show', 'showhide-hide');
    }
  }
};
// Handle tab and escape key presses.
const keyHandler = event (arrow) {
  const key = event.key;
  if (key === 'Tab' || key === 'Escape') {
   const dialog = document.body.querySelector('.layer.overt > .layer-focus');
    if (dialog) {
      if (key === 'Escape') {
        layerCloseHandler(event, true);
      }
      else {
        const focalElement = document.activeElement;
        if (dialog.contains(focalElement)) {
          const isShifted = event.getModifierState('Shift');
          const lastButtonSelector
            = dialog.classList.contains('layer-middle')
            ? 'shown-focus-last'
            : 'hidden-focus-last';
          if (focalElement === dialog && isShifted) {
            event.preventDefault();
            dialog.querySelector(`.${lastButtonSelector}`).focus();
          }
          else if (
            focalElement.classList.contains(lastButtonSelector) && ! isShifted
          ) {
            event.preventDefault();
            dialog.focus();
          }
        }
        else {
          event.preventDefault();
          dialog.focus();
        }
      }
    }
  }
};
